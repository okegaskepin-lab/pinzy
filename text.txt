    <script>
        // Cek login
        const userData = localStorage.getItem('web2apk_user');
        if (!userData) {
            window.location.href = 'login.html';
        }

        const user = JSON.parse(userData);
        const isOwner = user.username === 'Pinzy' || user.role === 'owner';

        // Tampilkan user info
        document.getElementById('userInfo').innerHTML = `
            <div class="user-avatar">${user.username.charAt(0).toUpperCase()}</div>
            <div>
                <div class="user-name">${user.username}</div>
                <div class="user-role">${isOwner ? 'Owner' : 'User'}</div>
            </div>
        `;

        // Tampilkan owner panel kalo owner
        if (isOwner) {
            document.getElementById('ownerPanel').style.display = 'block';
        }

        let selectedFile = null;
        let buildType = 'release';

        function selectBuildType(type) {
            buildType = type;
            document.querySelectorAll('.build-btn').forEach(btn => {
                btn.classList.remove('active');
                if (btn.dataset.type === type) btn.classList.add('active');
            });
        }

        function handleFileSelect(input) {
            if (input.files.length > 0) {
                selectedFile = input.files[0];
                document.getElementById('fileName').textContent = selectedFile.name;
            }
        }

        function startBuild() {
            if (!selectedFile) {
                alert('❌ Pilih file .zip dulu!');
                return;
            }
            if (!selectedFile.name.endsWith('.zip')) {
                alert('❌ Hanya file .zip yang diterima!');
                return;
            }

            document.getElementById('progressSection').style.display = 'block';
            
            const statuses = [
                'Getting Flutter dependencies...',
                'Running Gradle task...',
                'Building Flutter APK...',
                'Packaging APK...'
            ];
            
            let progress = 0;
            const interval = setInterval(() => {
                progress += Math.random() * 15;
                if (progress >= 100) {
                    progress = 100;
                    clearInterval(interval);
                    document.getElementById('progressStatus').textContent = '✅ Build Selesai! APK siap di-download.';
                    setTimeout(() => alert('✅ Build Sukses!\n\nAPK: ' + selectedFile.name.replace('.zip', '.apk')), 500);
                } else {
                    const idx = Math.floor((progress / 100) * statuses.length);
                    document.getElementById('progressStatus').textContent = statuses[Math.min(idx, statuses.length - 1)];
                }
                document.getElementById('progressFill').style.width = progress + '%';
            }, 500);
        }

        function resetForm() {
            selectedFile = null;
            document.getElementById('fileInput').value = '';
            document.getElementById('fileName').textContent = 'Belum ada file dipilih';
            document.getElementById('progressSection').style.display = 'none';
            document.getElementById('progressFill').style.width = '0%';
        }

        function broadcast() {
            const msg = prompt('📢 Masukkan pesan broadcast:');
            if (msg) alert('✅ Broadcast dikirim ke semua user!');
        }

        function viewUsers() {
            alert('👥 Premium Users:\n• @user1 - 15K Permanen\n• @user2 - 10K (exp 2 hari)\n• @user3 - 5K (exp 5 jam)');
        }

        function editPrices() {
            alert('💰 Edit Harga License:\n2K = 1 jam\n5K = 7 jam\n10K = 12 jam\n15K = Permanen');
        }

        function logout() {
            localStorage.removeItem('web2apk_user');
            window.location.href = 'login.html';
        }

        // Drag & drop
        const uploadArea = document.querySelector('.upload-area');
        uploadArea.addEventListener('dragover', (e) => {
            e.preventDefault();
            uploadArea.style.borderColor = '#00d4ff';
            uploadArea.style.background = 'rgba(0, 212, 255, 0.1)';
        });
        uploadArea.addEventListener('dragleave', () => {
            uploadArea.style.borderColor = '#2a4a7a';
            uploadArea.style.background = 'transparent';
        });
        uploadArea.addEventListener('drop', (e) => {
            e.preventDefault();
            uploadArea.style.borderColor = '#2a4a7a';
            uploadArea.style.background = 'transparent';
            const files = e.dataTransfer.files;
            if (files.length > 0) {
                selectedFile = files[0];
                document.getElementById('fileName').textContent = selectedFile.name;
            }
        });
    </script>
